function calculate(operator) { let num1 = document.getElementById("num1").value let num2 = document.getElementById("num2").value var ans; switch(operator) { case "+": //plus ans = num1 + num2 break; case "-": //minus ans = num1 - num2 break; case "*": // multiply ans = num1 * num2 break; case "/": // divide ans = num1 / num2 break; case "square root": // square root ans = Math.sqrt(num1) break; default: ans = "Please choose an operator" } return ans; }
